import os
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

path = r'D:\ML\data\ex1data1.txt'
data = pd.read_csv(path, header=None, names=['Population', 'Profit'])
data.head()

使用散点图显示数据

data.plot(kind='scatter', x='Population', y='Profit', figsize=(10, 6))

向数据集中添加新的一列1，方便线性回归的处理。对应假设函数的常数项 theta_0 对应的 x_0 (x_0默认为1)

data.insert(0, 'Ones', 1)
data.head()

cols = data.shape[1] #对应样例的特征数量（包括1这个新添加的特征值）
X = data.iloc[:, 0:cols-1] 
y = data.iloc[:, cols-1:cols]
#转换成矩阵，方便处理
X = np.matrix(X.values)
y = np.matrix(y.values)
theta = np.matrix(np.array([0, 0]))
#查看矩阵的维度
X.shape, y.shape, theta.shape

运行结果： ((97, 2), (97, 1), (1, 2))

下面先给出代价函数和梯度下降函数的定义：
$$ J(\Theta)=\frac{1}{2m}\sum_{i=1}^{m}(h_{\Theta}(x^{i})-y^{i})^{2} $$

Repeat until convergence:
$$ \Theta_{0}:=\Theta_{0}-a\frac{1}{m}\sum_{i=1}^{m}((h_{\Theta}(x^{(i)})-y^{(i)})x_{0}^{(i)}) $$

for j = 1,2,….n:

$$ \Theta_{j}:=\Theta_{j}-a\frac{1}{m}\sum_{i=1}^{m}(h_{\Theta}(x^{(i)})-y^{(i)})x_{j}^{(i)}) $$

#定义代价函数
def computeCost(X, y, theta):
    inner = np.power((X * theta.T) - y, 2)
    return np.sum(inner) / (2 * len(X))

#定义梯度下降函数
def gradientDescent(X, y, theta, alpha, iters):
    temp = np.matrix(np.zeros(theta.shape))
    parameters = theta.shape[1]
    cost = np.zeros(iters)

    for i in range(iters):
        error = (X * theta.T) - y

        for j in range(parameters):
            term = np.multiply(error, X[:,j])
            temp[0,j] = theta[0,j] - (alpha / len(X)) * np.sum(term)

        theta = temp
        cost[i] = computeCost(X, y, theta)

    return theta, cost

计算在初始 theta 条件下的代价函数值： 

computeCost(X, y, theta)

运行结果： 32.072733877455676

下面进行迭代2000次的梯度下降计算：

alpha = 0.01 #学习率
iters = 2000 #迭代次数

g, cost = gradientDescent(X, y, theta, alpha, iters) #g是迭代之后的theta值，cost保存每一次迭代对应的代价函数值

展示训练之后的拟合直线：

x = np.linspace(data.Population.min(), data.Population.max(), 100)
f = g[0, 0] + (g[0, 1] * x)
#fig 表示的是要绘制的一个画布，
#ax 表示的是在fig这个画布的对象上，具体要绘制的对象子图
fig, ax = plt.subplots(figsize=(10, 6)) #默认画布是1*1，当前使用的是第1个子图
ax.plot(x, f, 'r', label='Prediction') #绘制函数图像
ax.scatter(data.Population, data.Profit, label='Traning Data') #绘制散点图
ax.legend(loc=0) #使用0-10表示图例的位置
ax.set_xlabel('Population')
ax.set_ylabel('Price')
ax.set_title('Predicted Profit vs. Population Size')

展示训练次数和代价值的关系：

fig, ax = plt.subplots(figsize = (10, 6))
ax.plot(np.arange(iters), cost, 'r') #r表示红色
ax.set_xlabel('Iterations')
ax.set_ylabel('Cost')
ax.set_title('Error vs. Training Epoch')
